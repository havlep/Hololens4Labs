133 public void MakeScrollingList()
134 {
135 if (scrollView == null)
136 {
137 GameObject newScroll = new GameObject("Scrolling Object Collection");
138 newScroll.transform.parent = scrollPositionRef ? scrollPositionRef : transform;
139 newScroll.transform.localPosition = Vector3.zero;
140 newScroll.transform.localRotation = Quaternion.identity;
141 newScroll.SetActive(false);
142 scrollView = newScroll.AddComponent<ScrollingObjectCollection>();
143
144 // Prevent the scrolling collection from running until we're done dynamically populating it.
145 scrollView.CellWidth = cellWidth;
146 scrollView.CellHeight = cellHeight;
147 scrollView.CellDepth = cellDepth;
148 scrollView.CellsPerTier = cellsPerTier;
149 scrollView.TiersPerPage = tiersPerPage;
150 }
151
152 gridObjectCollection = scrollView.GetComponentInChildren<GridObjectCollection>();
153
154 if(gridObjectCollection != null)
155 {
156 Destroy(gridObjectCollection.gameObject);
157 }
158
159
160 GameObject collectionGameObject = new GameObject("Grid Object Collection");
161 collectionGameObject.transform.position = scrollView.transform.position;
162 collectionGameObject.transform.rotation = scrollView.transform.rotation;
163
164 gridObjectCollection = collectionGameObject.AddComponent<GridObjectCollection>();
165 gridObjectCollection.CellWidth = cellWidth;
166 gridObjectCollection.CellHeight = cellHeight;
167 gridObjectCollection.SurfaceType = ObjectOrientationSurfaceType.Plane;
168 gridObjectCollection.Layout = LayoutOrder.ColumnThenRow;
169 gridObjectCollection.Columns = cellsPerTier;
170 gridObjectCollection.Anchor = LayoutAnchor.UpperLeft;
171
172 scrollView.AddContent(collectionGameObject);
173
174 scrollView.gameObject.SetActive(true);
175
176 /*
177 if (!lazyLoad)
178 {
179 for (int i = 0; i < numItems; i++)
180 {
181 AddItemTest(itemPrefab);
182 }
183 scrollView.gameObject.SetActive(true);
184 gridObjectCollection.UpdateCollection();
185 }
186 else
187 {
188 if (loader != null)
189 {
190 loader.SetActive(true);
191 }
192
193 StartCoroutine(UpdateListOverTime(loader, itemsPerFrame));
194 }

198 private IEnumerator UpdateListOverTime(GameObject loaderViz, int instancesPerFrame)
199 {
200 for (int currItemCount = 0; currItemCount < numItems; currItemCount++)
201 {
202 for (int i = 0; i < instancesPerFrame; i++)
203 {
204 AddItemTest(itemPrefab);
205 }
206 yield return null;
207 }
208
209 // Now that the list is populated, hide the loader and show the list
210 loaderViz.SetActive(false);
211 scrollView.gameObject.SetActive(true);
212
213 // Finally, manually call UpdateCollection to set up the collection
214 gridObjectCollection.UpdateCollection();
215 }
