144 private void StartPhotoMode()
145 {
146 var cameraResolution = PhotoCapture.SupportedResolutions
147 .OrderByDescending((res) => res.width * res.height)
148 .First();
149
150 var cameraParams = new CameraParameters()
151 {
152 hologramOpacity = 0f,
153 cameraResolutionWidth = cameraResolution.width,
154 cameraResolutionHeight = cameraResolution.height,
155 pixelFormat = CapturePixelFormat.JPEG
156 };
157
158 photoCapture.StartPhotoModeAsync(cameraParams, startResult =>
159 {
160 Debug.Log($"Camera system start result = {startResult.resultType}.");
161 IsCameraActive = startResult.success;
162 onCameraStarted?.Invoke();
163 });
164 }

242 public Task<ImageData> TakePhotoWithThumbnail(Log log)
243 {
244 if (!IsCameraActive)
245 {
246 throw new CameraException("Can't take photo when camera is not ready.");
247 }
248
249 return Task.Run(() =>
250 {
251 var completionSource = new TaskCompletionSource<ImageData>();
252
253 AppDispatcher.Instance().Enqueue(() =>
254 {
255 Debug.Log("Starting photo capture.");
256
257#if UNITY_WSA
258 photoCapture.TakePhotoAsync((photoCaptureResult, frame) =>
259 {
260 Debug.Log("Photo capture done.");
261
262 var buffer = new List<byte>();
263 frame.CopyRawImageDataIntoBuffer(buffer);
264 var texture = new Texture2D(2, 2);
265 var data = buffer.ToArray();
266 texture.LoadImage(data);
267 var imageData = new ImageData(DateTime.Now, SceneController.CurrentUser, log);
268 imageData.Data = data;
269 imageData.Texture = texture;
270
271 completionSource.TrySetResult(imageData);
272 });
273#else
274 var tex = new Texture2D(webCamTexture.width, webCamTexture.height);
275 tex.SetPixels(webCamTexture.GetPixels());
276 tex.Apply();
277 var data = tex.EncodeToPNG();
278
279 var imageData = new ImageData(DateTime.Now, SceneController.CurrentUser, log);
280 imageData.Data = data;
281 imageData.Texture = tex;
282
283 completionSource.TrySetResult(imageData);
284#endif
285 });
286
287 return completionSource.Task;
288 });
289 }
