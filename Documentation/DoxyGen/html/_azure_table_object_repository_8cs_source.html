64 public async Task<OBJ> Create(OBJ obj)

69
70 var dto = mapper.CreateDTO(obj);
71

74
75 if ( dto.PartitionKey == null || dto.PartitionKey == string.Empty )
76 dto.PartitionKey = defaultPartitionKey;
77
78 // See if there is an object with a similiar id
79 var query = new TableQuery<DTO>().Where(
80 TableQuery.CombineFilters(
81 TableQuery.GenerateFilterCondition("PartitionKey", QueryComparisons.Equal, dto.PartitionKey),
82 TableOperators.And,
83 TableQuery.GenerateFilterCondition("RowKey", QueryComparisons.Equal, dto.RowKey)));
84 var segment = await this.table.ExecuteQuerySegmentedAsync(query, null);
85
86 var respDTO = segment.Results.FirstOrDefault();
87 if (respDTO != null)
88 // Object with a given id already exists throw an exception
89 throw new ObjectDataBaseException("Object of given ID" + dto.PartitionKey +" already exists");
90
91
92 // Insert the object into the database
93 var insertOrMergeOperation = TableOperation.InsertOrMerge(dto);
94 var res = await this.table.ExecuteAsync(insertOrMergeOperation);
95
96 if (res.Result == null)
97 throw new ObjectDataBaseException();

99 Debug.Log($"Object {dto.RowKey} created in repo");
