91 private async Task<bool> SetupTablesRepository()
92 {
93 cloudTableClient = storageAccount.CreateCloudTableClient();
94 experimentsTable = cloudTableClient.GetTableReference(experimentsTableName);
95 scientistsTable = cloudTableClient.GetTableReference(scientistsTableName);
96 logsTable = cloudTableClient.GetTableReference(logsTableName);
97
98 if (tryCreateTableOnStart)
99 {
100 try
101 {
102 if (await experimentsTable.CreateIfNotExistsAsync())
103 {
104 Debug.Log($"Created table {experimentsTableName}.");
105 }
106 if (await scientistsTable.CreateIfNotExistsAsync())
107 {
108 Debug.Log($"Created table {scientistsTableName}.");
109 }
110 if (await logsTable.CreateIfNotExistsAsync())
111 {
112 Debug.Log($"Created table {logsTableName}.");
113 }
114
115 }
116 catch (StorageException ex)
117 {
118 Debug.LogError("Failed to connect with Azure Storage.\nIf you are running with the default storage emulator configuration, please make sure you have started the storage emulator.");
119 Debug.LogException(ex);
120 onRepoReadyInitFailed?.Invoke();
121 }
122 }
123
124 // Setup the repository objects that will be used to get data
125 atExperimentRepository = new ATExperimentRepository(experimentsTable, partitionKey);
126 atScientistRepository = new ATScientistRepository(scientistsTable, partitionKey);
127 atLogRepository = new ATLogRepository(logsTable, partitionKey);
128 return true;
129
130 }

137 private bool SetupBlobRepository()

154 Debug.LogException(ex);
155 onRepoReadyInitFailed?.Invoke();
156 }
157 }
158 blobRepository = new AzureBlobRepository(blobContainer);
159 return true;

226 public async Task<Log> CreateLog(Log log)
227 {
228
229 if (log.Id == string.Empty)
230 log.Id = System.Guid.NewGuid().ToString();
231
232 if (log is TextLog
233 && (log as TextLog).TextData != null
234 && (log as TextLog).TextData.Id == string.Empty)
235 (log as TextLog).TextData.Id = System.Guid.NewGuid().ToString();
236
237 else if (log is ImageLog && (log as ImageLog).Data != null) {
238 var imageLog = log as ImageLog;
239
240 if (imageLog.Data.Id == string.Empty)
241 imageLog.Data.Id = System.Guid.NewGuid().ToString();
242
243 imageLog.Data.ThumbnailBlobName = await blobRepository.UploadBlob(imageLog.Data.Data, imageLog.Data.Id);
244
245 } else if(log is TranscriptionLog && (log as TranscriptionLog).Data != null)
246 {
247 var transcriptionLog = log as TranscriptionLog;
248
249 if (transcriptionLog.Data.Id == string.Empty)
250 transcriptionLog.Data.Id = System.Guid.NewGuid().ToString();
251
252 transcriptionLog.Data.ThumbnailBlobName = await blobRepository.UploadBlob(transcriptionLog.Data.Data, transcriptionLog.Data.Id);
253
254 }else
255 throw new System.NotImplementedException();
256
257 return await atLogRepository.Create(log);
258
259 }
